IN - укороченный вариант оператора OR/AND
Используется для минимизации множественных условий ИЛИ
Сравнвиает все значения, присутствующие внутри него
Выполняет сравнение между родительским и дочерним запросом
С помощью данного оператора можно проводить поиск по совпадениям букв и тд

WHERE имя_поля1 [NOT] IN (выражение)

Примеры:
SELECT * FROM имя_таблицы1
WHERE имя_поля1 IN (20, 23, 31, 40)
=
SELECT * FROM имя_таблицы1
WHERE имя_поля1 = 20
   OR имя_поля1 = 23
   OR имя_поля1 = 31
   OR имя_поля1 = 40;

SELECT * FROM имя_таблицы1
WHERE имя_поля1 IN (SELECT имя_поля2 FROM имя_таблицы2)
