1.int.Parse("343434") - преобразовать строку в число
2.int.TryParse("3434", out имя_переменной) - попытаться преобразовать и присвоить в переменную (выполняется при условии if  и тд)
- if (int.TryParse("3434", out имя_переменной)) {} else {} (out позволяет изменить значение переменной и выдать его наружу метода)
3.Convert.ToInt32("2323") - преобразовать строку в число
