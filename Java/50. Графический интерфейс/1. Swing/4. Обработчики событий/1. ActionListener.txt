ActionListener - слушатель, который срабатывает после отпускания мыши у компонента
В слушатель можно добавить уже готовый экшен, также можно добавить анонимный класс слушателя и переопределить его

компонент.addActionListener(new ActionListener() {
  @Override
  public void actionPerformed(ActionEvent e) {
    действие по нажатию кнопки
  }
});
