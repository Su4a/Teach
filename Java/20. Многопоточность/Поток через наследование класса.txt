Запуск потока через наследование от класса Тhread:

В большинстве случаев создавать подкласс, порождённый от класса Thread, следует только в случае, если требуется дополнить его новыми функциями

Создаем собственный класс, наследуем его от класса Thread, переопределяем метод run()

public class MyThread extends Thread {
	@Override
	public void run() {
		for (int i = 0; i < 5; i++) {
			System.out.println("Thread out: " + i);
		}
	}
}

public class MainApp {
	public static void main(String[] args) {
		Thread thread1 = new MyThread();
		Thread thread2 = new MyThread();
		thread1.start();
		thread2.start();
	}
}
